  def valid_input?
    if not valid_type?
      return false
    else
      Array(@@types[base_type]).each {|input_type| return true if input.is_a?(input_type)}
      raise MapReduceError, "invalid input (#{@input.inspect}) for type: #{base_type}. Try one of the following: #{Array(@@types[base_type]).join(", ")}"
    end
  end
